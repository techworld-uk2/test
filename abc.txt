Sub ChangeHeaderSequence()
    Dim sourceWorkbook As Workbook
    Dim targetWorkbook As Workbook
    Dim sourceSheet As Worksheet
    Dim targetSheet As Worksheet
    Dim sourceHeaderRange As Range
    Dim targetHeaderRange As Range
    Dim sourceCell As Range
    Dim targetCell As Range
    Dim headerName As String
    
    ' Set the source workbook (where the desired header sequence is defined)
    Set sourceWorkbook = Workbooks.Open("C:\Path\To\SourceWorkbook.xlsx")
    
    ' Set the target workbook (where the header sequence will be changed)
    Set targetWorkbook = Workbooks.Open("C:\Path\To\TargetWorkbook.xlsx")
    
    ' Set the source sheet (where the desired header sequence is defined)
    Set sourceSheet = sourceWorkbook.Sheets("SourceSheet")
    
    ' Set the target sheet (where the header sequence will be changed)
    Set targetSheet = targetWorkbook.Sheets("TargetSheet")
    
    ' Set the source header range (assuming it starts from cell A1)
    Set sourceHeaderRange = sourceSheet.Range("A1").CurrentRegion.Rows(1)
    
    ' Set the target header range (assuming it starts from cell A1)
    Set targetHeaderRange = targetSheet.Range("A1").CurrentRegion.Rows(1)
    
    ' Loop through each cell in the source header range
    For Each sourceCell In sourceHeaderRange
        ' Get the header name from the source cell
        headerName = sourceCell.Value
        
        ' Find the corresponding cell in the target header range with the same name
        Set targetCell = targetHeaderRange.Find(headerName, LookIn:=xlValues, LookAt:=xlWhole)
        
        ' If a matching cell is found, change its position in the target header range
        If Not targetCell Is Nothing Then
            targetCell.Cut
            targetHeaderRange.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromLeftOrAbove
        End If
    Next sourceCell
    
    ' Clear the source header range (optional, if needed)
    sourceHeaderRange.ClearContents
    
    ' Clear the target header range (optional, if needed)
    targetHeaderRange.ClearContents
    
    ' Save and close the workbooks
    sourceWorkbook.Close SaveChanges:=False
    targetWorkbook.Close SaveChanges:=True
EndÂ Sub
